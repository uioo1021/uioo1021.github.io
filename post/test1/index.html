放大缩小重置

let fzBtnCnt = 1;
const fzMin = 1; // 最小缩放比例
const fzMax = 1.5; // 最大缩放比例

fzBig.addEventListener('click', () => {
  if (fzBtnCnt < fzMax) {
    fzBtnCnt += 0.1; // 放大0.1
    main.style.transform = `scale(${fzBtnCnt})`;
    header.style.transform = `scale(${fzBtnCnt})`;
    footerDiv.style.transform = `scale(${fzBtnCnt})`;
    fz.style.transform = `scale(${fzBtnCnt})`;
    backgroundWT.style.transform = `scale(${fzBtnCnt})`;
    loginDiv.style.transform = `scale(${fzBtnCnt})`;
  }
});

fzSmall.addEventListener('click', () => {
  if (fzBtnCnt > fzMin) {
    fzBtnCnt -= 0.1; // 缩小0.1
    main.style.transform = `scale(${fzBtnCnt})`;
    header.style.transform = `scale(${fzBtnCnt})`;
    footerDiv.style.transform = `scale(${fzBtnCnt})`;
    fz.style.transform = `scale(${fzBtnCnt})`;
    backgroundWT.style.transform = `scale(${fzBtnCnt})`;
    loginDiv.style.transform = `scale(${fzBtnCnt})`;
  }
});

fzNew.addEventListener("click", () => {
  fzBtnCnt = 1; // 重置缩放比例

  footerDiv.style.transform = `scale(${fzBtnCnt})`;

vue3操作dom

<template>
  <div>
    <button @click="changeColor">点击我</button>
    <div ref="myDiv" class="box">这是一个盒子</div>
  </div>
</template>

<script setup>
import { ref } from 'vue'; 

// 使用 ref 获取 DOM 元素
const myDiv = ref(null);

// 改变盒子的颜色
const changeColor = () => {
  if (myDiv.value) {
    myDiv.value.style.backgroundColor = 'lightblue';
  }
};
</script>

}
